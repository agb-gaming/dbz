'''built in function'''
# Absolute Value
number = -10
print(f"Absolute value of {number}: {abs(number)}")

# Length of a String
text = "Hello, World!"
print(f"Length of the string '{text}': {len(text)}")

# Maximum and Minimum
numbers = [10, 20, 5, 40]
print(f"Maximum: {max(numbers)}, Minimum: {min(numbers)}")

# Sum of Elements
print(f"Sum of numbers: {sum(numbers)}")

# Type of a Variable
print(f"Type of 'numbers': {type(numbers)}")

# Convert to String
num = 123
print(f"Number as a string: {str(num)}")

# Round a Float
float_num = 5.678
print(f"Rounded value of {float_num}: {round(float_num, 2)}")
